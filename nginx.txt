server {
    listen 80;
    server_name web.bdsdaily.com;
    return 301 https://$server_name$request_uri;
}
server {
    server_name web.bdsdaily.com;
    listen 443 ssl http2;
    ssl_certificate /etc/letsencrypt/live/web.bdsdaily.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/web.bdsdaily.com/privkey.pem;

    root /var/www/bdsdaily/dist;
    index index.html;


    gzip on;
    gzip_comp_level 5;
    gzip_min_length 256;
    gzip_vary on;
    gzip_types
        text/plain
        text/css
        application/json
        application/javascript
        text/xml
        application/xml
        application/xml+rss
        text/javascript;

    location /api/ {
        proxy_pass https://app.bdsdaily.com/;
        proxy_set_header Host app.bdsdaily.com;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header X-Forwarded-Host $host;

        # Tăng timeout nếu cần (tùy backend)
        proxy_connect_timeout 60s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;

        # Bật buffering nếu backend trả về data lớn
        proxy_buffering on;
    }
    location / {
        try_files $uri $uri/ /index.html;
    }

    location = /index.html {
     add_header Cache-Control "no-cache, no-store, must-revalidate";
    }

    location /assets/ {
        expires 30d;
        add_header Cache-Control "public, immutable";
        access_log off;
    }

    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-XSS-Protection "1; mode=block" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;
}
